CodeChef - FORESTGA Forest Gathering（二分）

Posted on 2017-07-25 | In 二分

题目： https://vjudge.net/contest/173506#problem/F
题意：有n颗树，每颗有其初始高度和月长高量。现给出订单高度 w 以及限制砍树的高度 l（达到此高度才可砍掉）。问最少需要几个月才能完成订单。

分析：暴力超时，二分答案。细节看代码。

`C

#include

#include

#include

#include

#include

using namespace std;
const int maxn = 2e5 +7;
const int inf =0x3f3f3f3f;
typedef long long ll;
ll h[maxn],r[maxn],c[maxn],lim[maxn];
ll n,w,l;
bool check(ll t){
ll g=w;
for(ll i=1;i<=n;i++){
if(t>=c[i]){
if(t>=lim[i]) return 1;
else g-= h[i]+r[i]*t;
}
}
if(g<=0) return 1;
return 0;
}

int main()
{
freopen(“in.txt”,”r”,stdin);
cin>>n>>w>>l;
for(int i=1;i<=n;i++){
cin>>h[i]>>r[i];
c[i]=(l-h[i])/r[i];
if((l-h[i])%r[i] !=0) c[i]++;
lim[i]=(w-h[i])/r[i];
if((w-h[i])%r[i] !=0) lim[i]++;
}
ll st=0,ed=1e18,m;
while(st<ed){
m=(st+ed)>>1;
if(check(m)) ed=m;
else st=m+1;
}
if(check(st-1)) st–;
cout<<st;
return 0;
}
